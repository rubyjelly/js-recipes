Apply a function to each element in an array Demo 

	var charSets = new Array("ab","bb","cd","ab","cc","ab","dd","ab");
	
	function replaceElement(element,index,array) {
		if (element == "ab") array[index] = "**";
	}
	
	charSets.forEach(replaceElement);
	console.log(charSets[0]);

Use map() to convert an array into a new (non-referencing) one Demo 

	function convertValues(element,index,array) {
		var suffix = "a";
		element += suffix;
		return element;
	}
	
	var myArray = new Array(22,34,21,18,28,4);
	var newArray = myArray.map(convertValues);
	console.log(myArray);
	console.log(newArray);
	myArray[0] = 30;
	// original array[0] changed
	console.log(myArray[0] + "after");
	// new array did not change
	console.log(newArray[0] + "after");
